// Deposit button handler const depositBtn = document.getElementById('depositBtn'); depositBtn.addEventListener('click', function() { const depositAmount = getInputNumber("depositAmount"); updateSpan("currentDeposit", depositAmount); updateSpan("balanceStatus", depositAmount); document.getElementById('depositAmount').value = ""; }); // Withdrawal button handler const withdrawBtn = document.getElementById('withdrawBtn'); withdrawBtn.addEventListener('click', function() { const withdrawAmount = getInputNumber("withdrawAmount"); const currentBalance = parseFloat(document.getElementById("balanceStatus").innerText); // Check if there is enough balance for withdrawal if (withdrawAmount <= currentBalance) { updateSpan("currentWithdraw", withdrawAmount); updateSpan("balanceStatus", -1 * withdrawAmount); } else { alert("Insufficient balance for withdrawal!"); } document.getElementById('withdrawAmount').value = ""; }); // Function to retrieve input number function getInputNumber(idName) { const amount = document.getElementById(idName).value; return parseFloat(amount) || 0; // Return 0 if amount is not a valid number } // Function to update span with added number function updateSpan(idName, addedNumber) { console.log("ID name:", idName); // Log the ID name console.log("Added number:", addedNumber); // Log the added number const element = document.getElementById(idName); // Get the element by ID console.log("Element:", element); // Log the element if (element) { const current = parseFloat(element.innerText); const total = current + addedNumber; element.innerText = total.toFixed(2); // Ensure 2 decimal places } else { console.error("Element with ID", idName, "not found"); // Log error if element not found } }
